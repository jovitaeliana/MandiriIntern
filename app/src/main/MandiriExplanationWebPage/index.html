Project Overview

During my internship at Bank Mandiri, I developed a news application called Mandiri News. This Android application fetches real-time news from the News API and presents it in an intuitive, responsive interface.

The app was designed with user experience in mind, featuring a clean layout, responsive design, and smooth animations. It integrates modern Android development practices and libraries to deliver a seamless news browsing experience.

News API integration with Retrofit and custom OkHttpClient for API key authentication:

// NewsApiService.java
public interface NewsApiService {
    @GET("top-headlines")
    Call<NewsResponse> getTopHeadlines(
            @Query("country") String country,
            @Query("category") String category,
            @Query("q") String query,
            @Query("page") int page,
            @Query("pageSize") int pageSize
    );

    @GET("everything")
    Call<NewsResponse> searchAllNews(
            @Query("q") String query,
            @Query("language") String language
    );
}

// NewsRecyclerAdapter.java (excerpt)
@Override
public void onBindViewHolder(@NonNull NewsViewHolder holder, int position) {
    var article = articleList.get(position);
    holder.titleTextView.setText(article.getTitle());
    holder.sourceTextView.setText(article.getSource().getName());
    Picasso.get().load(article.getUrlToImage())
            .error(R.drawable.no_image_icon)
            .placeholder(R.drawable.no_image_icon)
            .into(holder.imageView);
    
    // Animation and click handling
    holder.itemView.setOnClickListener(v -> {
        var intent = new Intent(v.getContext(), NewsFullActivity.class);
        intent.putExtra("url", article.getUrl());
        v.getContext().startActivity(intent);
    });
}

Search Functionality

Expandable search view with real-time search capability:

// MainActivity.java (excerpt)
private void setupSearchView() {
    final int originalWidth = searchView.getLayoutParams().width;
    searchView.setIconified(true);

    searchView.setOnSearchClickListener(v -> {
        searchView.setLayoutParams(new LinearLayout.LayoutParams(
                LinearLayout.LayoutParams.MATCH_PARENT,
                LinearLayout.LayoutParams.WRAP_CONTENT
        ));
    });

    searchView.setOnQueryTextListener(new SearchView.OnQueryTextListener() {
        @Override
        public boolean onQueryTextSubmit(String query) {
            searchAllNews(query);
            searchView.clearFocus();
            return true;
        }

        @Override
        public boolean onQueryTextChange(String newText) {
            return false;
        }
    });
}

@Override
public void onClick(View v) {
    // Reset previous button color
    if (currentSelectedButton != null) {
        currentSelectedButton.setBackgroundColor(getResources().getColor(R.color.my_primary));
    }

    // Set new button color
    v.setBackgroundColor(getResources().getColor(R.color.my_secondary));
    currentSelectedButton = (Button) v;

    int id = v.getId();
    // Set category based on button ID
    if (id == R.id.btn_1) currentCategory = "general";
    else if (id == R.id.btn_2) currentCategory = "business";
    // Additional categories...

    // Reset page and fetch news
    currentPage = 1;
    getNews(currentCategory, null);
    recyclerView.smoothScrollToPosition(0);
}

Pagination

"View More" functionality with pagination:

// MainActivity.java (excerpt)
private void loadMoreNews() {
    if (isLoading) return;

    isLoading = true;
    btnViewMore.setVisibility(View.GONE);
    progressIndicator.setVisibility(View.VISIBLE);
    currentPage++;

    Call<NewsResponse> call = apiService.getTopHeadlines("us", currentCategory, null, currentPage, PAGE_SIZE);
    call.enqueue(new Callback<NewsResponse>() {
        @Override
        public void onResponse(Call<NewsResponse> call, Response<NewsResponse> response) {
            progressIndicator.setVisibility(View.INVISIBLE);
            isLoading = false;

            if (response.isSuccessful() && response.body() != null) {
                List<Article> newArticles = response.body().getArticles();
                if (newArticles != null && !newArticles.isEmpty()) {
                    int oldSize = articleList.size();
                    articleList.addAll(newArticles);
                    adapter.notifyItemRangeInserted(oldSize, newArticles.size());
                    btnViewMore.setVisibility(View.VISIBLE);
                }
            }
        }
        
        @Override
        public void onFailure(Call<NewsResponse> call, Throwable t) {
            // Error handling
        }
    });
}
